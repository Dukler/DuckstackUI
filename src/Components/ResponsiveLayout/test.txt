useEffect(() => {
    let auxFractions = {};
    let sumProps = 0;
    let count = 0;
    const length = responsiveArr.length;
    // const split = 100 / responsiveArr.length;
    const keys = Object.keys(multipliers);
    keys.forEach((key) => {
        if (multipliers[key]) {
            sumProps += multipliers[key];
            count++;
            auxFractions[key] = multipliers[key] / 100;
        }
    });
    if (sumProps <= 100) {
        if (count === length) {
            const arr = Object.entries(auxFractions);
            let min = Number.POSITIVE_INFINITY
            let key = "";
            for (const value of arr) {
                const aux = Math.min(min, value[1]);
                key = aux !== min ? value[0] : key;
                min = aux;
            }
            multipliers[key] = false;
            count--;
            sumProps = sumProps - (min * 100)
        }
        const responsiveCount = length - count;
        const rest = sumProps
        const calc = ((100 - rest) / responsiveCount) / 100;

        keys.forEach((key) => {
            if (!multipliers[key]) {
                auxFractions[key] = calc;
            }
            else {
                auxFractions[key] += length - count;
            }
        });
    } else {
        console.log("mas de 100")
    };
    setMultis(auxFractions);
}, [multipliers, responsiveArr.length]);
